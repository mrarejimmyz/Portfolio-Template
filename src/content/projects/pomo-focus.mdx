from PyQt5.QtCore import QTimer

class PomodoroTimer:
def **init**(self):

self.time_left = self.duration
self.timer = QTimer()
self.timer.timeout.connect(self.update_timer)

    def start(self):
        self.timer.start(1000)  # Update every second

    def update_timer(self):
        self.time_left -= 1
        if self.time_left <= 0:
            self.timer.stop()
            self.on_timer_complete()
